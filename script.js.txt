// الرابط المباشر لجدولك
const csvUrl = "https://docs.google.com/spreadsheets/d/e/2PACX-1vSdht6CoTy4Ldy3IguEeGB3c4lHKjnorQwyJe0D6BWG10t9D21BB-Q_-JZilVd8m6aslgNMYy93u2Yi/pub?output=csv";

async function fetchAndGenerate() {
    const orderInput = document.getElementById('orderInput');
    const userInput = document.getElementById('userInput');

    if (!orderInput || !userInput) {
        alert("تأكد أن أسماء الخانات في HTML هي orderInput و userInput");
        return;
    }

    const orderId = orderInput.value.trim();
    const username = userInput.value.trim().toLowerCase();

    try {
        const response = await fetch(csvUrl);
        if (!response.ok) throw new Error("فشل الوصول لجدول جوجل، تأكد من النشر بصيغة CSV");
        
        const data = await response.text();
        const rows = data.split(/\r?\n/).map(row => row.split(","));

        let secret = null;
        // البحث عن السر في العمود C
        for (let i = 1; i < rows.length; i++) {
            if (rows[i][0]?.trim() === orderId && rows[i][1]?.trim().toLowerCase() === username) {
                secret = rows[i][2]?.trim();
            }
        }

        if (secret) {
            document.getElementById('resultArea').style.display = 'block';
            startGenerator(secret);
        } else {
            alert("لم نجد بيانات مطابقة! تأكد من رقم الطلب واسم الحساب في الجدول.");
        }
    } catch (e) {
        alert("مشكلة في الاتصال: " + e.message);
    }
}

function startGenerator(s) {
    if (typeof SteamTotp === 'undefined') {
        alert("محرك الأكواد (Library) لم يتم تحميله! تأكد من وجود سطر المكتبة في الـ HTML");
        return;
    }
    
    const update = () => {
        try {
            const code = SteamTotp.generateAuthCode(s);
            document.getElementById('steamCode').innerText = code;
            let sec = 30 - (Math.floor(Date.now() / 1000) % 30);
            document.getElementById('timerText').innerText = `يتغير خلال: ${sec} ثانية`;
        } catch (err) {
            console.error("خطأ في السر:", err);
        }
    };
    update();
    if (window.timer) clearInterval(window.timer);
    window.timer = setInterval(update, 1000);
}
